    airports[['iata_code', 'latitude_deg', 'longitude_deg', 'elevation_ft']],
    left_on='Origin',
    right_on='iata_code',
    how='left'

# Join 2: BTS + Weather (daily)
weather_daily = weather.groupby(['station', 'date']).agg({
    'vsby': 'mean', 'tmpf': 'mean', 'p01i': 'sum'
}).reset_index()

merged = merged.merge(
    weather_daily,
    left_on=['Origin', 'FlightDate'],
    right_on=['station', 'date'],
    how='left'
)

# Join 3: BTS + Reviews (monthly aggregation)
reviews['carrier_code'] = reviews['airline_name'].map(mapping)
reviews['year_month'] = pd.to_datetime(reviews['date_flown']).dt.to_period('M')

review_agg = reviews.groupby(['carrier_code', 'year_month']).agg({
    'overall_rating': 'mean'
}).reset_index()

merged['year_month'] = merged['FlightDate'].dt.to_period('M')
merged = merged.merge(
    review_agg,
    left_on=['Carrier', 'year_month'],
    right_on=['carrier_code', 'year_month'],
    how='left'
)
